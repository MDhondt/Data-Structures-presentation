\section{Introductory Data Structures}

\begin{frame}
	\frametitle{What are Data Structures?}
	
	\begin{mydef}{Data Structure}
		A way in which data are stored for efficient search and retrieval. Different data structures are suited for different problems.\footnote[frame]{\tiny{Encyclop\ae dia Britannica}}
	\end{mydef}
	
	\vspace{5mm}
	
	\begin{itemize}
		\item $\text{Data type} \neq \text{data structure}$
		\item \texttt{java.util.HashSet} vs. hash table
		\item \texttt{array} vs. array
	\end{itemize}
\end{frame}

	\begin{definition}
		A linked list is a data structure in which the objects are arranged in a linear order. Unlike arrays in which the linear order is determined by indices, the order is determined by a pointer in each object.\footnote[frame]{\tiny{Introduction to Algorithms By Cormen, Leierson, Rivest \& Stein}}
	\end{definition}
	
	\begin{itemize}
		\item Different types: singly, doubly, multiply, circular, \ldots
		\item Example: doubly linked list
	\end{itemize}

\deeptocsubsection{Tree}

\begin{frame}
	\frametitle{Tree}

\subsubsection{Heap}

\begin{frame}
	\frametitle{Binary Heap}
	
	\begin{definition}[Heap]
		A complete tree where every node has a key more extreme (greater or less) than or equal to the key of its parent.\footnote[frame]{\tiny{National Institute of Standards \& Technology}}
	\end{definition}
	
	\begin{definition}[Binary Heap]
		A binary heap data structure is an array object that we can view as a nearly complete binary tree that satisfies the min-heap or max-heap property.\footnote[frame]{\tiny{Introduction to Algorithms By Cormen, Leierson, Rivest \& Stein}}
	\end{definition}
\end{frame}

\begin{frame}
	\frametitle{Bloom filter}
	
	\begin{block}{Math}{\begin{multicols}{2}\begin{itemize}
		\item $m$: array length
		\item $k$: \# hash functions
		\item $n$: \# of inserted elements
		\item $p$: false positive probability
	\end{itemize}\end{multicols}\vspace{-2mm}
	\begin{itemize}
		\item \# bits needed: $m=\displaystyle\frac{-n\ln{p}}{(\ln{2})^2}$
		\item Optimal \# of hash functions: $k=\displaystyle\frac{m}{n}\ln(2)$
	\end{itemize}}\end{block}
	
	\begin{itemize}
		\item<2-> Supports unions and intersections
		\item<2-> Fits arbitrarily \# of elements (false positive rate increases)
		\item<2-> Common application: caching
	\end{itemize}
	\vspace{1cm}
	
\end{frame}

	\frametitle{van Emde Boas Tree}
	
	\begin{operations}\begin{itemize}
		\item \parbox{7cm}{\texttt{min}, \texttt{max}, \texttt{isEmpty}}$O(1)$
		\item \parbox{7cm}{\texttt{insert}, \texttt{delete}, \texttt{contains}, \texttt{succ}, \texttt{pred}}$O(\log\log U)$
	\end{itemize}\end{operations}
	
	\begin{itemize}
		\item $O(U)$ space
		\begin{itemize}
			\item X-fast trie: $O(n\log U)$ space
			\item Y-fast trie: $O(n)$ space
		\end{itemize}
		\item $U = 2^k, k\in\mathbb{N}$
	\end{itemize}
\end{frame}
